      $n$&$n!$&$n^2$&$e^n$\\
      \hline\hline
      {clj>
        (do
          (require '[clojure.math :as math])
          (defn factorial [x]
            (apply *' (range 2 (inc x))))
          (for [n (range 11)
                :let [f (factorial n)
                      d (int (math/pow n 2))
                      e (format "%.2f" (math/exp n))]]
            (str n "&" f "&" d "&" e "\\\\ \\hline")))
      >clj}
    \end{tabular}
